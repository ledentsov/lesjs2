d = (2+ b++); alert(d); // 4 получаем так как в b = 1 второго первого действия и инкрементирование происходит после присвоением значения + 2
alert(a); // 3 после первого действия а=2(инкремент 1+1) после 3го действия a=3(инкремент2+1)
alert(b); // 3 после второго действия b=2(инкремент 1+1) после 4го действия b=3(инкремент2+1)
